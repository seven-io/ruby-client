def initialize(resource)
  Sms77::Util::get_namespace_classes(Sms77::Resources).each do |cls|
    name = cls.name.split('::').last

    instance_variable_set("@#{name}", cls.new(resource))

    singleton_class.instance_eval("attr_reader :#{name}")
  end
end
